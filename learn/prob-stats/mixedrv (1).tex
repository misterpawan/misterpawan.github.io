\begin{align*}
f_X(x) = \begin{cases}

\end{cases}

\begin{align*}
Y = g(X) = \begin{cases}

\end{cases}
\end{align*}
Find the CDF of $Y.$
\end{alertblock}

\subsection{CDF of mixed RV}



\begin{frame}{CDF of mixed RV as a sum of continuous and discrete RV...}
\pause 

\begin{itemize}
\item In the previous example, CDF of $Y$ was \pause 
\begin{align*}
F_Y(y) = \begin{cases}
1 \quad &y \geq 1/2 \\
y^2 \quad &0 \leq y < 1/2 \\
0 \quad &\text{otherwise}
\end{cases}
\end{align*} \pause 
\item The CDF can be written as a sum of continuous and discrete part \pause 
\begin{align*}
F_Y(y) = C(y) + D(y),
\end{align*} \pause 
where the continuous part is \pause 
\begin{align*}
C(y) = \begin{cases}
1/4 \quad &y \geq 1/2 \\
y^2 \quad &0 \leq y < 1/2 \\
0 \quad &y < 0
\end{cases}
\end{align*} \pause 
and the discrete part is \pause  
\begin{align*}
D(y) = \begin{cases}
3/4 \quad &y \geq 1/2 \\
0 \quad &y<1/2
\end{cases}
\end{align*}
\end{itemize}

\end{frame}

\begin{frame}{CDF of a mixed RV as a sum of Continuous and Discrete CDF...}
\pause 

\begin{alertblock}{CDF of Mixed Random Variable}
\pause 

The CDF of mixed random variable $Y$ can be written as a sum of continuous and discrete/staircase function $$F_Y(y) = C(y) + D(y).$$ \pause We differentiate the continuous part $c(y) = \dfrac{dC(y)}{dy},$ where ever $C(y)$ is differentiable. \pause Let $\{ y_1, y_2, \dots, \}$ be the set of jump points of $D(y),$ i.e., the points for which $P(Y=y_k)>0.$ \pause  We have 
\begin{align*}
\int_{-\infty}^{\infty} c(y) \, dy + \sum_{y_k} P(Y = y_k) = 1
\end{align*} \pause 
The expected value of $Y$ can be obtained as  \pause 
\begin{align*}
E[Y] = \int_{-\infty}^{\infty} y \, c(y) \, dy + \sum_{y_k} y_k \,  P(Y = y_k)
\end{align*}
\end{alertblock}

\end{frame}
